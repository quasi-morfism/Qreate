## Reference Configuration

1) vite.config.js must configure base path to support sub-path deployment, need to support importing files via @, don't configure port number


import { defineConfig } from 'vite'
import vue from '@vitejs/plugin-vue'

export default defineConfig({
  base: './',
  plugins: [vue()],
  resolve: {
    alias: {
      '@': fileURLToPath(new URL('./src', import.meta.url))
    }
  }
})


2) Router configuration must use hash mode to avoid server-side routing configuration issues

import { createRouter, createWebHashHistory } from 'vue-router'

const router = createRouter({
  history: createWebHashHistory(),
  routes: [
    // Route configuration
  ]
})
